### コード実行ブロック（Python3）
LLMの前に配置。現在日時から最短の体験予約可能枠を自動計算する。

```python
import datetime

def main() -> dict:
    JST = datetime.timezone(datetime.timedelta(hours=9))
    now = datetime.datetime.now(JST)
    today = now.date()
    current_time = now.time()
    
    weekdays_jp = ['月', '火', '水', '木', '金', '土', '日']
    
    # 体験可能時間（16:00は削除済み）
    weekday_slots = ["18:00", "20:00"]           # 月〜金
    saturday_slots = ["10:00", "14:00", "18:00"] # 土
    # 日曜・祝日は定休日
    
    holidays = [
        datetime.date(2026, 1, 1),
        datetime.date(2026, 1, 12),
        datetime.date(2026, 2, 11),
        datetime.date(2026, 2, 23),
        datetime.date(2026, 3, 20),
        datetime.date(2026, 4, 29),
        datetime.date(2026, 5, 3),
        datetime.date(2026, 5, 4),
        datetime.date(2026, 5, 5),
        datetime.date(2026, 5, 6),
        datetime.date(2026, 7, 20),
        datetime.date(2026, 8, 11),
        datetime.date(2026, 9, 21),
        datetime.date(2026, 9, 23),
        datetime.date(2026, 10, 12),
        datetime.date(2026, 11, 3),
        datetime.date(2026, 11, 23),
    ]
    
    candidates = []
    
    for i in range(14):
        d = today + datetime.timedelta(days=i)
        weekday = d.weekday()
        
        if weekday == 6:
            continue
        if d in holidays:
            continue
        
        if weekday == 5:
            slots = saturday_slots
        else:
            slots = weekday_slots
        
        for slot in slots:
            slot_hour = int(slot.split(":")[0])
            slot_minute = int(slot.split(":")[1])
            
            if d == today:
                deadline = datetime.time(slot_hour - 1, slot_minute)
                if current_time >= deadline:
                    continue
            
            day_str = f"{d.month}月{d.day}日（{weekdays_jp[weekday]}）{slot}"
            candidates.append(day_str)
    
    nearest = candidates[:3] if len(candidates) >= 3 else candidates
    
    today_str = f"{today.year}年{today.month}月{today.day}日（{weekdays_jp[today.weekday()]}）{now.strftime('%H:%M')}現在"
    
    return {
        "today": today_str,
        "nearest_slots": "\n".join(nearest),
        "all_slots": "\n".join(candidates[:6]),
    }
```

出力変数：
- `today`（string）— 今日の日付と現在時刻
- `nearest_slots`（string）— 最短3枠
- `all_slots`（string）— 6枠分
